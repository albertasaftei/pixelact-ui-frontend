import { PixelInput, PixelLabel } from "pixelact-ui";
import "pixelact-ui/styles.css";

## PixelInput

A pixel art styled input component with retro styling and full support for all HTML input attributes. Features pixel font styling and custom visual design.

### Basic Input

<CodePreview 
  language="tsx"
  code={`import { PixelInput } from "pixelact-ui";

export function BasicInput() {
  return (
    <PixelInput 
      placeholder="Enter your text..."
      type="text"
    />
  );
}`}
>
  <PixelInput className="w-full" placeholder="Enter your text..."  />
</CodePreview>

### With Label
