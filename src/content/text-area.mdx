import { PixelTextArea } from "pixelact-ui";

## PixelTextArea

A pixel art styled textarea component built on top of shadcn/ui's Textarea component. Features retro styling with pixel font and full support for all HTML textarea attributes.

### Basic TextArea

<CodePreview 
  language="tsx"
  code={`import { PixelTextArea } from "pixelact-ui";

export function BasicTextArea() {
  return (
    <PixelTextArea 
      placeholder="Enter your message..."
      rows={4}
    />
  );
}`}
>
  <PixelTextArea placeholder="Enter your message..." rows={4} />
</CodePreview>
